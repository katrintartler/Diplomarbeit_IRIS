#include <Arduino.h>
#include <ArduinoJson.h>

HardwareSerial SerialESP(2); // notwendig für RX2 und TX2

// ESP_A Code
const int sensors[6] = {25, 26, 32, 33, 34, 35};

void setup() {
  Serial.begin(115200); 
  SerialESP.begin(115200, SERIAL_8N1, 16, 17); // 16 = RXD2, 17 = TXD2
  // RX2 und TX2 verwenden, weil RX1 und TX1 intern probleme machen

  // ADC konfigurieren
  analogReadResolution(12);       // 0-4095, 12 Bit Auflösung
  analogSetAttenuation(ADC_11db); // voller Messbereich

  // ADC initialisieren
  for (int i = 0; i < 6; i++) {
    pinMode(sensors[i], INPUT);
  }
}

void loop() {
  static unsigned long lastSend = 0;
  
  if (millis() - lastSend > 1000) {
    int v[6];

    for (int i = 0; i < 6; i++) {
      // v[i] = analogRead(sensors[i]);
      long average = 0;

      for (int a = 0; a < 10; a++) {
        average += analogRead(sensors[i]);
        delayMicroseconds(200); // verhindert schnelle ADC Spikes
      }

      v[i] = average/10;
    }

    // JSON Dokument erstellen
    StaticJsonDocument<256> doc;

    // Struktur
    doc["version"] = 2;
    doc["timestamp"] = millis();

    JsonObject movement = doc.createNestedObject("movement");

    movement["s1_lu"] = v[0];   // links unten
    movement["s1_lm"] = v[1];   // links mitte
    movement["s1_lo"] = v[2];   // links oben
    movement["s1_ru"] = v[3];   // rechts unten
    movement["s1_rm"] = v[4];   // rechts mitte
    movement["s1_ro"] = v[5];   // rechts oben

    String json;
    serializeJson(doc, json);   // wandelt das Objekt in einen String

    SerialESP.println(json);

    lastSend = millis();
  }
}

  switch (type) {
    case WStype_CONNECTED:
      Serial.println("PWA verbunden");

    case WStype_DISCONNECTED:
      Serial.println("PWA getrennt");
      break;

    case WStype_TEXT:
      Serial.print("Von PWA empfangen: ");
      Serial.println((char*)payload);
      break;
  }
}

void setup() {
  Serial.begin(115200);
  Serial2.begin(115200, SERIAL_8N1, 17, 16); // 17 = RXD2, 16 = TXD2

  // Kontrollleuchte
  pinMode(connectionLED, OUTPUT);
  digitalWrite(connectionLED, LOW);

  // Wifi verbinden
  WiFi.begin(ssid, password);
  Serial.print("Aufbau der WLAN Verbindung");

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWLAN verbunden!");
  digitalWrite(connectionLED, HIGH); // Kontrollleuchte für die WLAN Verbindung
  Serial.print("ESP IP: "); // 172.20.10.5
  Serial.println(WiFi.localIP());

  // WebSocket starten
  webSocket.begin();
  webSocket.onEvent(webSocketEvent);

  Serial.println("WebSocket Server gestartet (Port 81)");
}

void loop() {
  if (millis() - lastWifiCheck > checkInterval) {
    lastWifiCheck = millis();

    if (WiFi.status() != WL_CONNECTED) {
      Serial.println("WiFi Verbindung getrennt -> Verbindungsaufbau wird versucht ...");
      digitalWrite(connectionLED, LOW);
      WiFi.disconnect();
      WiFi.begin(ssid, password);
    } 
    else {
      digitalWrite(connectionLED, HIGH);
    }
  }

  webSocket.loop();

  // Daten von ESP_A empfangen und an PWA senden
  if (Serial2.available()) {
    String data = Serial2.readStringUntil('\n');
    webSocket.broadcastTXT(data);
  }
}

let ws;

const reconnectInterval = 2000; // 2 Sekunden Reconnect Interval

function connectWebSocket() {
    if (ws) { // falls noch eine alte Version da ist, wird diese geschlossen
        ws.onclose = null;
        ws.close();
    }

    ws = new WebSocket("ws://172.20.10.6:81");  // IP von ESP B - localhost Anwendung
    //ws = new WebSocket("ws://192.168.8.189:3000") // Laptop IP - Server Anwendung

    ws.onopen = () => {
        console.log("Verbunden!");
        updateStatus(true);

        if (reconnectTimeout) {
            clearTimeout(reconnectTimeout);
            reconnectTimeout = null;
        }
    };

    ws.onmessage = (event) => {
        // console.log("Daten vom ESP: ", event.data);

        try {
            // JSON.parse() macht aus dem text von event.data ein JavaScriptObjekt
            const data = JSON.parse(event.data); // bedeutet man kann jetzt auf die Sensoren zugreifen also z.B. data.movement.s1_lu
            const movementData = data.movement;

            // Bewegungswert berechnen
            const movementValue = deltaMovement(movementData);

            console.log("Bewegungswert: ", movementValue);

            // Movement Status
            const statusMov = getStatusMov(movementValue);

            updateMovementStatus(statusMov);    // Status
            updateValues(movementData);         // Sensorwerte
            updateChart(movementValue);         // Delta Werte
        }
        catch (e) {
            console.error("JSON Fehler: ", e);
        }/* 
        catch ( ... ) {

        }*/
    };

    ws.onclose = () => {
        console.log("Verbindung verloren! - neuer Verbindungsversuch ...");
        updateStatus(false);

        if (!reconnectTimeout) {
            reconnectTimeout = setTimeout(() => {
                connectWebSocket();
            }, reconnectInterval);
        }

        /* Version 2 Code
        setTimeout(() => {
            connectWebSocket();
        }, reconnectInterval); */
    };

    ws.onerror = (err) => {
        console.error("WebSocket Fehler: ", err);
        ws.close(); // wichtig, weil dadurch wird onclose getriggert
    };
}

function updateStatus(isConnected) {
    const statusElement = document.getElementById("status");

    if (!statusElement) return;

    if (isConnected) {
        statusElement.textContent = "Verbunden!";
        statusElement.style.color = "green";
    }
    else {
        statusElement.textContent = "Getrennt!";
        statusElement.style.color = "red";
    }
}

function deltaMovement(current) {
    if(!lastValues) { // lastValues definieren, falls es nicht definiert ist
        lastValues = current;
        return 0;
    }

    let deltaSum = 0;

    for (let key in current) {
        let delta = Math.abs(current[key] - lastValues[key]); // Math.abs = Betrag
        deltaSum += delta; // Änderungen werden addiert
    }

    lastValues = current;
    return deltaSum;
}

function getStatusMov(movementValue) {
    if (movementValue < 50) {
        return {text: "Ruhig", color: "green"};
    }
    else if (movementValue < 300) {
        return {text: "Leichte Bewegung", color: "orange"};
    }
    else {
        return {text: "Starke Bewgung",  color: "red"};
    }
}

function updateMovementStatus(statusObject) {
    // document = ganzes html Dokument
    const el = document.getElementById("movementStatus");

    el.textContent = statusObject.text;
    el.style.backgroundColor = statusObject.color;
    el. style.color = "white";
    el.style.padding = "10px";  // Innenabstand
    el.style.borderRadius = "10px"; // abgerundete Ecken
}

function updateValues(values) {
    const container = document.getElementById("values");
    let html = "";

    for (let key in values) {
        html +=`<p>${key}: ${values[key]}</p>`;
    }

    container.innerHTML = html;
}

function updateChart(value) {
    if (!chart) return; // Guard Clause - solange chart nicht existiert wird die Funktion beendet

    chart.data.labels.push("");
    chart.data.datasets[0].data.push(value); // push hängt die neuen Daten hinten an

    // Chart zeigt die letzten 120 Werte
    if (chart.data.labels.length > 120) {  
        chart.data.labels.shift(); // shift() ... entfernt das erste Element
        chart.data.datasets[0].data.shift();
    }

    chart.update();
}

async def ble_connect():
    global client 
    client = BleakClient(ESP_MAC)
    await client.connect()
    print("client connected") 


def ble_start_thread():
    def run():
        asyncio.set_event_loop(ble_loop)
        ble_loop.run_until_complete(ble_connect())
        print("BLE loop running")
        ble_loop.run_forever()
    
    threading.Thread(target=run,daemon=True).start()

  CMD_IDLE,
  CMD_TWITCHING,
  CMD_GM,
  CMD_LM_L_LEFT,
  CMD_LM_L_RIGHT,
}t_state_commands; 

typedef enum {
  ENTRY,
  DURING,
}t_state_executionstate; 

t_states states = IDLE; 
t_state_commands state_commands = CMD_IDLE; 
t_state_executionstate state_executionstate;

void BleCmd(){
  if (tw_flag){
    tw_flag = false; 
    state_commands = CMD_TWITCHING; 
  }
  if (gm_flag){
    gm_flag = false; 
    state_commands = CMD_GM; 
  }
  if (lm_l_flag){
    lm_l_flag = false; 
    state_commands = CMD_LM_L_LEFT; 
  }
  if (lm_r_flag){
    lm_r_flag = false; 
    state_commands = CMD_LM_L_RIGHT; 
  }
  if(crying_flag){
    myPlayer.volume(10);
    myPlayer.play(1);
    crying_flag = false; 
  }
  if(brabbeln_flag){
    brabbeln_flag = false; 
    myPlayer.volume(10);
    myPlayer.play(2);
  }
  if(coughing_flag){
    coughing_flag = false; 
    myPlayer.volume(10);
    myPlayer.play(3);
  }
  if(sneezing_flag){

void Servo1TargetReachedHandler(ServoEasing *aServoEasingInstance){
  switch(states){
    // was muss der Servo1 also Linkes Bein machen bei Twitching, wir gehen nach dem LUT
    // Servo 1 winkel sind in der ersten Spalte des LUT geschrieben 

    case TWITCHING:
      if (counter1 < SEQUENCES){
        Servo1.startEaseTo(TWITCHING_lut[counter1][0]);
        counter1++; 
        Serial.print("hallo");
        state_commands = CMD_IDLE; 
      }
      else {
        states = IDLE; 
        counter1 = 0; 
      }
    break; 

    case GM:
       if(counter1 < SEQUENCES){
        Servo1.startEaseTo(GM_lut[counter1][0]); 
        counter1++; 
        state_commands = CMD_IDLE; 
       }
       else{
        states = IDLE; 
        counter1 = 0; 
       }
    break; 

    case LM_L_LEFT: 
      if(counter1 < SEQUENCES){
        Servo1.startEaseTo(LM_L_LEFT_lut[counter1]); 
        counter1++; 
        state_commands = CMD_IDLE; 
      }
      else{
        states = IDLE; 
        counter1 = 0; 
      }
    break; 
    }
}

void ServoSM() {

  switch (states) {

    case IDLE:

    switch (state_commands) {

      case CMD_IDLE:
      break; 

      case CMD_TWITCHING:
      states = TWITCHING; 
      state_executionstate = ENTRY; 
      break; 
      
      case CMD_GM:
      states = GM;
      state_executionstate = ENTRY;
      break; 

      case CMD_LM_L_LEFT:
      states = LM_L_LEFT; 
      state_executionstate = ENTRY; 
      break; 

      case CMD_LM_L_RIGHT: 
      states = LM_L_RIGHT; 
      state_executionstate = ENTRY; 
      break; 

      default:
      break; 
    }
    break; 
    
    case TWITCHING:
    if(state_executionstate == ENTRY){
      Serial.print("in Tw");
      state_executionstate = DURING;
      Servo1TargetReachedHandler(&Servo1); 
      Servo2TargetReachedHandler(&Servo2); 
      Servo3TargetReachedHandler(&Servo3); 
      Servo4TargetReachedHandler(&Servo4); 
      // fehtl Servo3 und Servo4
    }
    break; 

    case GM: 
    if(state_executionstate == ENTRY){
      state_executionstate = DURING; 
      Servo1TargetReachedHandler(&Servo1); 
      Servo2TargetReachedHandler(&Servo2); 
    }
    break; 

    case LM_L_LEFT:
    if(state_executionstate == ENTRY){
      state_executionstate = DURING; 
      Servo1TargetReachedHandler(&Servo1); 
    }
    break; 

    case LM_L_RIGHT: 
    if (state_executionstate == ENTRY){
      state_executionstate = DURING; 
      Servo2TargetReachedHandler(&Servo2); 
    }
    break; 

    default:
    break; 
  }
}
